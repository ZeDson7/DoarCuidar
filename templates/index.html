from flask import render_template, request from app import app from app.consulta_empresas import buscar_empresas @app.route('/buscar_instituicoes', methods=['POST']) def buscar_instituicoes(): client_id = "SEU_CLIENT_ID" client_secret = "SEU_SECRET" palavra_chave = request.form.get('palavra_chave') estado = request.form.get('estado') try: empresas = buscar_empresas(client_id, client_secret, palavra_chave, estado) return render_template('resultados.html', empresas=empresas) except Exception as e: return render_template('resultados.html', erro=str(e))
